TActionRegister = record

  Dialog  : Template; // input/output: Action with parent link
  Action  : pointer;  // LastResult, hContact, Parameter

  Load    : pointer;  // no parameters
  Export,
  Import  : pointer;  // format number/code
  DataSize: integer;  // SizeOf Action type record
end;
